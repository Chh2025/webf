Exception Handling Examples in JavaScript

1. Handle Exception Caused by Undeclared Variable

try {
  console.log(x); // x is not declared
} catch (error) {
  console.log("Error caught:", error.message);
}

2. Throw "Divide by Zero" Error using throw Statement

function divide(a, b) {
  if (b === 0) {
    throw new Error("Divide by Zero Error");
  }
  return a / b;
}

try {
  let result = divide(10, 0);
  console.log("Result:", result);
} catch (error) {
  console.log("Caught:", error.message);
}

3. Throw "Out of Bound" Error using throw Statement

function getElement(arr, index) {
  if (index < 0 || index >= arr.length) {
    throw new Error("Out of Bound Error");
  }
  return arr[index];
}

try {
  let myArray = [10, 20, 30];
  console.log(getElement(myArray, 5));
} catch (error) {
  console.log("Caught:", error.message);
}

4. Exception Handling using finally Statement

try {
  let a = 5;
  let b = 0;
  if (b === 0) {
    throw new Error("Cannot divide by zero");
  }
  console.log(a / b);
} catch (error) {
  console.log("Error:", error.message);
} finally {
  console.log("This block always runs (cleanup, logging, etc.)");
}
